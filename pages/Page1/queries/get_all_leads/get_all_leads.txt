SELECT
    leads.id AS "מזהה_ליד",
    leads.updated_at AS "תאריך",
    customers.full_name AS "שם_הלקוח",
    customers.identity_number AS "תעודת_זהות",
    partners.name AS "מקור_הליד",
    leads.main_status AS "סטטוס",
    COALESCE(SUM(transactions.premium_amount), 0) AS "פרמיה_סך_הכל"
FROM leads
LEFT JOIN customers ON leads.customer_id = customers.id
LEFT JOIN partners ON leads.partner_id = partners.id
LEFT JOIN transactions ON leads.id = transactions.lead_id
WHERE
    -- חיפוש חופשי (שם או תעודת זהות) - ILIKE עבור כל תנאי חיפוש
    (customers.full_name ILIKE '%{{inp_search.text}}%'
     OR customers.identity_number ILIKE '%{{inp_search.text}}%'
     OR '{{inp_search.text}}' = '')
GROUP BY leads.id, leads.updated_at, customers.full_name, customers.identity_number, partners.name, leads.main_status
ORDER BY leads.updated_at DESC
